Mini Terminal en C

Nota: Este codigo es de las notas de la materia.

Descripción: Mini terminal que implementa comandos del sistema como pwd, ls, cd, cat, mkdir, stat, date, who, free, ip, mac y más.

Comandos disponibles:

  • pwd — Muestra el directorio actual
  • cd <directorio> — Cambia de directorio
  • ls [-l|-a|-i] — Lista contenido del directorio
  • mkdir <directorio> — Crea un nuevo directorio
  • cat <archivo> — Muestra el contenido de archivos
  • rm <archivo> — Elimina archivos
  • stat <archivo> — Muestra metadatos de archivo
  • date [-u] — Muestra la fecha y hora
  • who — Usuarios conectados
  • free [-h] — Uso de memoria del sistema
  • ip — Direcciones IP de tarjetas de red
  • mac — Direcciones MAC de tarjetas de red
  • exit — Salir de la terminal

typedef void (*HandlerComando)(char **args);

typedef struct {
    char *nombre;
    HandlerComando funcion;
    char *descripcion;
} Comando;

int main(int argc, char *argv[]) {
    (void)argc;
    (void)argv;

    Comando tabla_comandos[] = {
        {"pwd",    pwd,                 "Muestra el directorio actual"},
        {"cd",     function_cd,         "Cambia de directorio"},
        {"mkdir",  function_mkdir,      "Crea un nuevo directorio"},
        {"ls",     function_ls,         "Lista contenido (-l, -a, -i, -la, -li, -lai)"},
        {"cat",    function_cat,        "Muestra el contenido de uno o mas archivos"},
        {"rm",     function_rm,         "Elimina uno o mas archivos"},
        {"unlink", function_unlink_cmd, "Elimina uno o mas archivos (unlink)"},
        {"rename", function_rename,     "Renombra o mueve un archivo"},
        {"stat",   function_stat,       "Muestra metadatos de archivo (stat)"},
        {"vfstat", function_vfstat,     "Muestra metadatos usando fstat"},
        {"uname",  function_uname,      "Muestra informacion del sistema"},
        {"find",   function_find,       "Busca archivo en un solo nivel (sin recursion)"},
        {"date",   function_date,        "Muestra la fecha y hora actual"},
        {"who",    function_who,         "Muestra usuarios conectados"},
        {"mesg",   function_mesg,        "Habilita o deshabilita mensajes en el terminal"},
        {"wall",   function_wall,        "Envia un mensaje a todos los usuarios"},
        {"free",   function_free,        "Muestra uso de memoria del sistema"},
        {"ip",    function_ip,           "Muestra las direcciones IP de las tarjetas de red"},
        {"mac",   function_mac,          "Muestra las direcciones MAC de las tarjetas de red"},
        {NULL,     NULL,                NULL}
    };

    char entrada[200];
    char *parametros[64];

    printf("MiniShell Iniciada:\n");

    while (1) {
        printf("$ ");
        if (fgets(entrada, sizeof(entrada), stdin) == NULL) break;

        entrada[strcspn(entrada, "\n")] = 0;

        char *token = strtok(entrada, " \t");
        int i = 0;
        while (token != NULL && i < 63) {
            parametros[i++] = token;
            token = strtok(NULL, " \t");
        }
        parametros[i] = NULL;

        if (parametros[0] != NULL) {
            if (strcmp(parametros[0], "exit") == 0) break;

            int encontrado = 0;
            for (int j = 0; tabla_comandos[j].nombre != NULL; j++) {
                if (strcmp(parametros[0], tabla_comandos[j].nombre) == 0) {
                    tabla_comandos[j].funcion(parametros);
                    encontrado = 1;
                    break;
                }
            }
            if (!encontrado) {
                printf("Comando no encontrado: %s\n", parametros[0]);
            }
        }
        printf("\n");
    }
    return 0;
}

// Helpers
static void permisos_a_cadena(mode_t mode, char out[11]) {
    out[0] = S_ISDIR(mode) ? 'd' : S_ISLNK(mode) ? 'l' : '-';
    out[1] = (mode & S_IRUSR) ? 'r' : '-';
    out[2] = (mode & S_IWUSR) ? 'w' : '-';
    out[3] = (mode & S_IXUSR) ? 'x' : '-';
    out[4] = (mode & S_IRGRP) ? 'r' : '-';
    out[5] = (mode & S_IWGRP) ? 'w' : '-';
    out[6] = (mode & S_IXGRP) ? 'x' : '-';
    out[7] = (mode & S_IROTH) ? 'r' : '-';
    out[8] = (mode & S_IWOTH) ? 'w' : '-';
    out[9] = (mode & S_IXOTH) ? 'x' : '-';
    out[10] = '\0';
}

void function_cd(char *parametros[]) {
    const char *destino = parametros[1];
    if (destino == NULL) {
        destino = getenv("HOME");
        if (destino == NULL) {
            fprintf(stderr, "cd: HOME no esta definido\n");
            return;
        }
    }
    if (chdir(destino) == -1) perror("cd");
}

void function_stat(char *parametros[]) {
    if (parametros[1] == NULL) {
        fprintf(stderr, "Error: stat requiere al menos un archivo\n");
        return;
    }

    for (int i = 1; parametros[i] != NULL; i++) {
        struct stat st;
        if (stat(parametros[i], &st) == -1) {
            perror(parametros[i]);
            continue;
        }

        char perms[11], tbuf[64];
        permisos_a_cadena(st.st_mode, perms);
        struct tm *tm_info = localtime(&st.st_mtime);
        if (tm_info) strftime(tbuf, sizeof(tbuf), "%Y-%m-%d %H:%M:%S", tm_info);
        else snprintf(tbuf, sizeof(tbuf), "N/A");

        printf("Archivo: %s\n", parametros[i]);
        printf("  Inodo: %llu\n", (unsigned long long)st.st_ino);
        printf("  Permisos: %s\n", perms);
        printf("  Links: %lu\n", (unsigned long)st.st_nlink);
        printf("  UID:GID: %u:%u\n", st.st_uid, st.st_gid);
        printf("  Tamano: %lld bytes\n", (long long)st.st_size);
        printf("  Modificado: %s\n\n", tbuf);
    }
}

void function_vfstat(char *parametros[]) {
    if (parametros[1] == NULL) {
        fprintf(stderr, "Error: vfstat requiere al menos un archivo\n");
        return;
    }

    for (int i = 1; parametros[i] != NULL; i++) {
        int fd = open(parametros[i], O_RDONLY);
        if (fd == -1) {
            perror(parametros[i]);
            continue;
        }

        struct stat st;
        if (fstat(fd, &st) == -1) {
            perror("fstat");
            close(fd);
            continue;
        }

        char perms[11], tbuf[64];
        permisos_a_cadena(st.st_mode, perms);
        struct tm *tm_info = localtime(&st.st_mtime);
        if (tm_info) strftime(tbuf, sizeof(tbuf), "%Y-%m-%d %H:%M:%S", tm_info);
        else snprintf(tbuf, sizeof(tbuf), "N/A");

        printf("Archivo(fd): %s\n", parametros[i]);
        printf("  Inodo: %llu\n", (unsigned long long)st.st_ino);
        printf("  Permisos: %s\n", perms);
        printf("  Links: %lu\n", (unsigned long)st.st_nlink);
        printf("  UID:GID: %u:%u\n", st.st_uid, st.st_gid);
        printf("  Tamano: %lld bytes\n", (long long)st.st_size);
        printf("  Modificado: %s\n\n", tbuf);

        close(fd);
    }
}

void function_unlink_cmd(char *parametros[]) {
    if (parametros[1] == NULL) {
        fprintf(stderr, "Error: unlink requiere al menos un archivo\n");
        return;
    }

    for (int i = 1; parametros[i] != NULL; i++) {
        if (unlink(parametros[i]) == -1) perror(parametros[i]);
        else printf("unlink: '%s' eliminado\n", parametros[i]);

}

void function_uname(char *parametros[]) {
    struct utsname u;
    if (uname(&u) == -1) {
        perror("uname");
        return;
    }

    if (parametros[1] && strcmp(parametros[1], "-a") == 0) {
        printf("%s %s %s %s %s\n",
               u.sysname, u.nodename, u.release, u.version, u.machine);
    } else {
        printf("%s\n", u.sysname);
    }
}

void function_date(char *parametros[]) {
    int flag_u = (parametros[1] && strcmp(parametros[1], "-u") == 0);
    time_t t = time(NULL);
    struct tm *tm_info = flag_u ? gmtime(&t) : localtime(&t);
    char buf[64];
    strftime(buf, sizeof(buf), "%a %b %d %H:%M:%S %Z %Y", tm_info);
    printf("%s\n", buf);
}

void function_who(char *parametros[]) {
    (void)parametros;
    setutent();
    struct utmp *entry;
    while ((entry = getutent()) != NULL) {
        if (entry->ut_type == USER_PROCESS) {
            char tbuf[32];
            time_t t = (time_t)entry->ut_tv.tv_sec;
            struct tm *tm_info = localtime(&t);
            if (tm_info) strftime(tbuf, sizeof(tbuf), "%Y-%m-%d %H:%M", tm_info);
            else snprintf(tbuf, sizeof(tbuf), "N/A");
            printf("%-12s %-10s %s (%s)\n",
                   entry->ut_user, entry->ut_line, tbuf, entry->ut_host);
        }
    }
    endutent();
}

void function_mesg(char *parametros[]) {
    int fd = open(ttyname(STDIN_FILENO), O_RDONLY);
    if (fd == -1) { perror("mesg"); return; }
    struct stat st;
    if (fstat(fd, &st) == -1) { perror("mesg"); close(fd); return; }
    close(fd);

    if (parametros[1] == NULL) {
        printf("mesg is %s\n", (st.st_mode & S_IWGRP) ? "y" : "n");
    } else if (strcmp(parametros[1], "y") == 0) {
        if (chmod(ttyname(STDIN_FILENO), st.st_mode | S_IWGRP) == -1) perror("mesg");
        else printf("mesg: mensajes habilitados\n");
    } else if (strcmp(parametros[1], "n") == 0) {
        if (chmod(ttyname(STDIN_FILENO), st.st_mode & ~S_IWGRP) == -1) perror("mesg");
        else printf("mesg: mensajes deshabilitados\n");
    } else {
        fprintf(stderr, "Uso: mesg [y|n]\n");
    }
}

void function_wall(char *parametros[]) {
    if (parametros[1] == NULL) {
        fprintf(stderr, "Uso: wall <mensaje>\n");
        return;
    }

    char mensaje[512];
    mensaje[0] = '\0';
    for (int i = 1; parametros[i] != NULL; i++) {
        if (i > 1) strncat(mensaje, " ", sizeof(mensaje) - strlen(mensaje) - 1);
        strncat(mensaje, parametros[i], sizeof(mensaje) - strlen(mensaje) - 1);
    }

    pid_t pid = fork();
    if (pid == -1) { perror("wall"); return; }
    if (pid == 0) {
        char *args[] = {"wall", mensaje, NULL};
        execvp("wall", args);
        perror("wall");
        _exit(1);
    }
    int status;
    waitpid(pid, &status, 0);
}

void function_free(char *parametros[]) {
    int flag_h = (parametros[1] && strcmp(parametros[1], "-h") == 0);
    FILE *f = fopen("/proc/meminfo", "r");
    if (!f) { perror("free"); return; }

    long total = 0, libre = 0, disponible = 0, buffers = 0, cached = 0,
         swap_total = 0, swap_libre = 0;
    char linea[128], clave[64];
    long valor;

    while (fgets(linea, sizeof(linea), f)) {
        if (sscanf(linea, "%63s %ld", clave, &valor) == 2) {
            if      (strcmp(clave, "MemTotal:")     == 0) total       = valor;
            else if (strcmp(clave, "MemFree:")      == 0) libre       = valor;
            else if (strcmp(clave, "MemAvailable:") == 0) disponible  = valor;
            else if (strcmp(clave, "Buffers:")      == 0) buffers     = valor;
            else if (strcmp(clave, "Cached:")       == 0) cached      = valor;
            else if (strcmp(clave, "SwapTotal:")    == 0) swap_total  = valor;
            else if (strcmp(clave, "SwapFree:")     == 0) swap_libre  = valor;
        }
    }
    fclose(f);

    long usado = total - libre - buffers - cached;
    long swap_usado = swap_total - swap_libre;

    if (flag_h) {
        printf("%-10s %8s %8s %8s %8s %8s\n",
               "", "total", "usado", "libre", "buf/cach", "disponible");
        printf("%-10s %7ldM %7ldM %7ldM %7ldM %7ldM\n",
               "Mem:",
               total/1024, usado/1024, libre/1024,
               (buffers+cached)/1024, disponible/1024);
        printf("%-10s %7ldM %7ldM %7ldM\n",
               "Swap:",
               swap_total/1024, swap_usado/1024, swap_libre/1024);
    } else {
        printf("%-10s %10s %10s %10s %10s %10s\n",
               "", "total", "usado", "libre", "buf/cach", "disponible");
        printf("%-10s %10ld %10ld %10ld %10ld %10ld\n",
               "Mem:",
               total, usado, libre, buffers+cached, disponible);
        printf("%-10s %10ld %10ld %10ld\n",
               "Swap:",
               swap_total, swap_usado, swap_libre);
    }
}

void function_ip(char *parametros[]) {
    (void)parametros;
    int fd = socket(AF_INET, SOCK_DGRAM, 0);
    if (fd == -1) { perror("ip: socket"); return; }

    struct ifconf ifc;
    char buf[8192];
    ifc.ifc_len = sizeof(buf);
    ifc.ifc_buf = buf;

    if (ioctl(fd, SIOCGIFCONF, &ifc) == -1) {
        perror("ip: ioctl SIOCGIFCONF");
        close(fd);
        return;
    }

    int n = ifc.ifc_len / (int)sizeof(struct ifreq);
    printf("%-16s  %s\n", "Interfaz", "Direccion IP");
    printf("%-16s  %s\n", "--------", "------------");

    for (int i = 0; i < n; i++) {
        struct ifreq ifr;
        strncpy(ifr.ifr_name, ifc.ifc_req[i].ifr_name, IFNAMSIZ - 1);
        ifr.ifr_name[IFNAMSIZ - 1] = '\0';

        if (ioctl(fd, SIOCGIFADDR, &ifr) == -1) continue;

        struct sockaddr_in *addr = (struct sockaddr_in *)&ifr.ifr_addr;
        char ip_str[INET_ADDRSTRLEN];
        inet_ntop(AF_INET, &addr->sin_addr, ip_str, sizeof(ip_str));
        printf("%-16s  %s\n", ifr.ifr_name, ip_str);
    }

    close(fd);
}

    for (int i = 0; i < n; i++) {
        struct ifreq ifr;
        strncpy(ifr.ifr_name, ifc.ifc_req[i].ifr_name, IFNAMSIZ - 1);

    }

void function_find(char *parametros[]) {
    if (parametros[1] == NULL) {
        fprintf(stderr, "Uso: find <nombre_archivo> [directorio]\n");
        return;
    }

    const char *nombre = parametros[1];
    const char *base = (parametros[2] != NULL) ? parametros[2] : ".";

    DIR *dir = opendir(base);
    if (!dir) {
        perror("find");
        return;
    }

    struct dirent *ent;
    int total = 0;
    char ruta[4096];

    while ((ent = readdir(dir)) != NULL) {
        if (strcmp(ent->d_name, nombre) == 0) {
            if (strcmp(base, "/") == 0) snprintf(ruta, sizeof(ruta), "/%s", ent->d_name);
            else snprintf(ruta, sizeof(ruta), "%s/%s", base, ent->d_name);

            printf("\033[0;32m[Encontrado]: %s\033[0m\n", ruta);
            total++;
        }
    }

    closedir(dir);

    if (total == 0) {
        printf("\033[0;31mNo se encontraron coincidencias para: %s\033[0m\n", nombre);
    } else {
        printf("Busqueda finalizada. Se encontraron %d coincidencia(s).\n", total);
    }
}

void function_ls(char *parametros[]) {
    int flag_l = 0, flag_a = 0, flag_i = 0;
    const char *ruta = ".";

    for (int k = 1; parametros[k] != NULL; k++) {
        if (parametros[k][0] == '-') {
            for (int j = 1; parametros[k][j] != '\0'; j++) {
                if (parametros[k][j] == 'l') flag_l = 1;
                else if (parametros[k][j] == 'a') flag_a = 1;
                else if (parametros[k][j] == 'i') flag_i = 1;
                else {
                    fprintf(stderr, "ls: opcion invalida -- '%c'\n", parametros[k][j]);
                    return;
                }
            }
        } else {
            ruta = parametros[k];
        }
    }

    DIR *d = opendir(ruta);
    if (!d) {
        perror("ls");
        return;
    }

    struct dirent *e;
    char fullpath[4096];

    while ((e = readdir(d)) != NULL) {
        if (!flag_a && e->d_name[0] == '.') continue;

        if (strcmp(ruta, "/") == 0) snprintf(fullpath, sizeof(fullpath), "/%s", e->d_name);
        else snprintf(fullpath, sizeof(fullpath), "%s/%s", ruta, e->d_name);

        struct stat st;
        if (lstat(fullpath, &st) == -1) {
            perror(e->d_name);
            continue;
        }

        if (flag_i) printf("%8llu ", (unsigned long long)st.st_ino);

        if (flag_l) {
            char perms[11], tbuf[32];
            permisos_a_cadena(st.st_mode, perms);

            struct tm *tm_info = localtime(&st.st_mtime);
            if (tm_info) strftime(tbuf, sizeof(tbuf), "%b %d %H:%M", tm_info);
            else snprintf(tbuf, sizeof(tbuf), "N/A");

            printf("%s %3lu %8lld %s %s\n",
                   perms, (unsigned long)st.st_nlink, (long long)st.st_size, tbuf, e->d_name);
        } else {
            printf("%s\n", e->d_name);
        }
    }

    closedir(d);
}

void function_cat(char *parametros[]) {
    if (parametros[1] == NULL) {
        printf("Error: cat requiere un archivo\n");
        return;
    }

    char buf[1024];
    ssize_t r;

    for (int i = 1; parametros[i] != NULL; i++) {
        int fd = open(parametros[i], O_RDONLY);
        if (fd == -1) {
            perror(parametros[i]);
            continue;
        }

        struct stat s;
        if (fstat(fd, &s) == -1) {
            perror("cat fstat");
            close(fd);
            continue;
        }

        if (S_ISDIR(s.st_mode)) {
            fprintf(stderr, "cat: %s: es directorio\n", parametros[i]);
            close(fd);
            continue;
        }

        while ((r = read(fd, buf, sizeof(buf))) > 0) {
            (void)write(STDOUT_FILENO, buf, (size_t)r);
        }

        close(fd);
    }
}
