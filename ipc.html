Practica #3

Mecanismos de comunicacion entre procesos (IPC)

IPC pipe/fifo System V shm/msg

Objetivo

Implementar y comparar mecanismos IPC para intercambio y sincronizacion de datos entre procesos: tuberias, semaforos, memoria compartida y cola de mensajes, usando llamadas del sistema en Linux.

3. Mecanismos de comunicacion entre procesos - IPC

IPC permite que dos o mas procesos cooperen compartiendo datos o eventos. La eleccion del mecanismo depende del tipo de comunicacion (flujo continuo, mensajes discretos o acceso a memoria compartida).

3.1 Comunicacion mediante tuberias

La comunicación entre procesos es fundamental para que intercambien datos, para llevar a cabo dicha comunicación, hay que tener en consideración, en primer lugar, si los procesos se van a comunicar en la misma máquina y si están emparentados, y segundo, si estos va a comunicar desde máquinas diferentes.

3.1.1 Tuberias sin nombre - pipe

Codigo Fuente

#include <stdio.h>
    #include <stdlib.h>
    #include <sys/types.h>
#include <unistd.h>
#include <sys/wait.h>

    #define MAXLINEA 80

    int main()
    {
      int n, fd[2];
      pid_t hijo;
      char linea[MAXLINEA];

      if (pipe(fd) < 0) {
        fprintf(stderr, "error de pipe");
        exit(0);
      }

      if ((hijo = fork()) < 0) {
        fprintf(stderr, "error de fork");
        exit(EXIT_FAILURE);
      }
      else if (hijo > 0)
      { /* padre */
        close(fd[0]);
        write(fd[1], "hola mundo\n", 12);
        wait(NULL);
      }
      else {
        close(fd[1]);
        n = read(fd[0], linea, MAXLINEA);
        write(STDOUT_FILENO, linea, n);
      }

      return EXIT_SUCCESS;
}

Salida en Terminal

user@utm:~$ gcc -Wall tuberia_sin_nombre.c -o tuberia_sin_nombre
    user@utm:~$ ./tuberia_sin_nombre
    hola mundo

Lo que aprendi y como se podria mejorar

Aprendi que una pipe sin nombre solo funciona entre procesos relacionados y transmite bytes en una sola direccion.

Se podria mejorar cerrando siempre los extremos correctos, validando pipe() y fork() y manejando multiples mensajes.

3.1.2 Tuberias con nombre - fifo

Codigo Fuente

#include <stdio.h>
    #include <stdlib.h>
    #include <sys/types.h>
    #include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>

    int main (void)
    {
      pid_t hijo;
      int file;
      char mensaje[20];

      unlink("namepipe"); // borra el archivo del sistema de archivos
      umask(~0666); // cambia la mascara de permisos por defecto

      if(mkfifo("namepipe",0666)==-1)
      {
        perror("error en mkfifo");
        exit(EXIT_FAILURE);
      }

      if ( (hijo=fork())==0)
      {
        fprintf (stdout,"soy el hijo, ID=%ld\n",(long)getpid());
        if( (file=open("namepipe",O_WRONLY) ) ==-1)
        {
          perror("error en open O_WRONLY");
          exit(EXIT_FAILURE);
        }
        write (file,"soy el hijo,ID...\n",18);
        close(file);
        getchar();
        exit(EXIT_SUCCESS);
    }

      if (hijo > 0)
      {
        fprintf (stdout,"soy el padre, ID = %ld\n",(long)getpid());
        if( (file=open("namepipe",O_RDONLY)) == -1)
        {
          perror("error en open O_RDONLY");
          exit(EXIT_FAILURE);
        }
        int n = read(file,mensaje,20);
        if (n >= 0) {
          if (n < 20) mensaje[n] = '\0';
          else mensaje[19] = '\0';
        }
        fprintf(stdout,"%s\n",mensaje);
        close(file);
      }

      return EXIT_SUCCESS;
}

Lo que aprendi y como se podria mejorar

Aprendi que una FIFO permite comunicar procesos no emparentados a traves del sistema de archivos.

Se podria mejorar manejando mejor los bloqueos al abrir la tuberia, limpiando el archivo FIFO al terminar y validando la longitud leida.

3.2 Mecanismos IPC derivados de System V

El paquete de comunicación entre procesos de los sistema UNIX System V y derivados, como GNU/Linux, se componen de tres mecanismos: 1. Los semáforos, que permiten sincronizar procesos; 2. La memoria compartida, que va a permitir que los procesos compartan su espacio de direcciones virtuales; y 3. Las colas de mensajes, que posibilitan el intercambio de datos con un formato determinado.

3.2.2 Semaforos en derivados de System V

Codigo Fuente

Nota: Este codigo es de las notas de la materia.

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/ipc.h>
#include <sys/sem.h>
#include <sys/wait.h>

/* Estructura necesaria para semctl */
union semun {
    int val;
};

int main(int argc, char *argv[]) {

    key_t llave;
    int semid;
    pid_t pid;
    struct sembuf op;
    union semun arg;

    /* Generar clave IPC */
    llave = ftok(argv[0], 65);

    /* Crear semáforo */
    semid = semget(llave, 1, 0666 | IPC_CREAT);

    /* Inicializar semáforo en 0 */
    arg.val = 0;
    semctl(semid, 0, SETVAL, arg);

    pid = fork();

    if(pid == 0) {

        /* PROCESO HIJO */

        printf("Hijo: esperando señal del padre...\n");

        op.sem_num = 0;
        op.sem_op = -1;  // operación P (wait)
        op.sem_flg = 0;

        semop(semid, &op, 1);

        printf("Hijo: señal recibida, continuando ejecución.\n");

    } else {

        /* PROCESO PADRE */

        sleep(2);
        printf("Padre: enviando señal al hijo...\n");

        op.sem_num = 0;
        op.sem_op = 1;   // operación V (signal)
        op.sem_flg = 0;

        semop(semid, &op, 1);

        wait(NULL);

        /* Eliminar semáforo */
        semctl(semid, 0, IPC_RMID);

        printf("Padre: proceso terminado.\n");
    }

    return EXIT_SUCCESS;
}

Salida en Terminal

user@utm:~$ gcc -Wall semaforo_padre_hijo.c -o semaforo_padre_hijo
user@utm:~$ ./semaforo_padre_hijo
Hijo: esperando señal del padre...
Padre: enviando señal al hijo...
Hijo: señal recibida, continuando ejecución.
Padre: proceso terminado.

Lo que aprendi y como se podria mejorar

Aprendi a sincronizar padre e hijo con semaforos System V usando operaciones P y V.

Se podria mejorar validando todas las llamadas de semget(), semctl() y semop(), ademas de liberar recursos en caso de error.

3.3 Memoria compartida

Codigo Fuente

Nota: Este codigo es externo de las notas de la materia.

#include <stdio.h>
#include <string.h>
#include <sys/ipc.h>
#include <sys/shm.h>

int main(void) {
    key_t key = ftok("/tmp", 'M');
    int shmid = shmget(key, 128, IPC_CREAT | 0666);
    char *mem = (char *)shmat(shmid, NULL, 0);

    strcpy(mem, "Dato escrito en memoria compartida");
    printf("Lectura: %s\n", mem);
    return 0;
}

Salida en Terminal

user@utm:~$ ./shm_demo

Lo que aprendi y como se podria mejorar

Aprendi que la memoria compartida permite intercambiar datos rapidamente entre procesos.

Se podria mejorar liberando la memoria con shmdt() y shmctl(), y verificando errores de shmat() y shmget().

3.4 Cola de mensajes

Codigo Fuente

Nota: Este codigo es de las notas de la materia.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>
#include <unistd.h>
#include <errno.h>
#include <sys/msg.h>
void send_msg(int,int );
void get_msg(int, int);
struct msgbuf {
 long mtype;
 char mtext[80];
 };
int main(int argc, char *argv[])
{
 int qid;
 int modo; // 1 = enviar/escribir, 2 = recibir/leer
 int msgtype = 1;
 int llave;
 llave=ftok (argv[0],'a');
 if (argc>1)
 {
 if (strcmp(argv[1],"s")==0) modo=1;
 else if (strcmp(argv[1],"r")==0) modo=2;
 else { printf ("Use: ./mcola s|r\n");
 exit(EXIT_FAILURE);
 }
 } else {
 printf ("Use: ./mcola s|r\n");
 exit(EXIT_FAILURE);
 }
 // Crea cola de mensajes
 if ( (qid = msgget(llave, IPC_CREAT | 0666))==-1)
 {
 perror("msgget");
 exit(EXIT_FAILURE);
 }
 if (modo == 2) get_msg(qid, msgtype);
 else send_msg(qid, msgtype);
 return(EXIT_SUCCESS);
 }
void send_msg(int qid, int msgtype)
{
 struct msgbuf msg;
 time_t t;
 msg.mtype = msgtype;
 time(&t);
 snprintf(msg.mtext, sizeof(msg.mtext), "El mensaje salió el: %s", ctime(&t));
 // Enviar/Escribir en la cola
 if (msgsnd(qid, (void *) &msg, sizeof(msg.mtext), IPC_NOWAIT) == -1)
 {
 perror("ERROR en msgsnd");
 exit(EXIT_FAILURE);
 }
 printf("Mensaje enviado: %s\n", msg.mtext);
 }
void get_msg(int qid, int msgtype)
{
 struct msgbuf msg;
// Recibir/Leer de la cola
if (msgrcv(qid,(void*)&msg,sizeof(msg.mtext),msgtype,MSG_NOERROR|
IPC_NOWAIT) == -1) 
{
 if (errno != ENOMSG)
 {
 perror("ERROR en msgrcv");
 exit(EXIT_FAILURE);
 }
 printf("No hay mensajes disponibles para leer/recibir con msgrcv()\n");
 } else printf("Mensaje recibido: %s\n", msg.mtext);
 }
