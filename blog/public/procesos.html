Practica #2

Procesos e Hilos: llamadas al sistema y ejecucion concurrente

C fork wait/waitpid pthreads

Objetivo

Entender como el sistema operativo crea, identifica y sincroniza procesos, y comparar ese modelo con la creacion de hilos en un mismo espacio de memoria.

2.1 Introduccion a procesos

Un proceso es una instancia de un programa en ejecucion. Cada proceso tiene PID, estado, memoria y recursos propios. El sistema operativo planifica estos procesos para dar sensacion de paralelismo.

2.2 Practica 1: Arbol de procesos con fork() y waitpid()

Codigo Fuente

Nota: Este codigo es de las notas de la materia.

#include <stdio.h>
    #include <stdlib.h>
    #include <unistd.h>
    #include <sys/wait.h>

    void crear_arbol(int nivel_actual, int max_nivel) {

      printf("%*sProceso: PID = %d, Padre: PID = %d (Nivel %d)\n",
           nivel_actual * 4, "", getpid(), getppid(), nivel_actual);
      fflush(stdout);

      if (nivel_actual < max_nivel) {
        pid_t hijo1 = fork();
        if (hijo1 == 0) {
          // Proceso hijo 1
          crear_arbol(nivel_actual + 1, max_nivel);
          exit(0);
        } else if (hijo1 > 0) {
          pid_t hijo2 = fork();
          if (hijo2 == 0) {
            // Proceso hijo 2
            crear_arbol(nivel_actual + 1, max_nivel);
            exit(0);
          } else if (hijo2 > 0) {
            // Proceso padre espera a ambos hijos
            waitpid(hijo1, NULL, 0);
            waitpid(hijo2, NULL, 0);

            sleep(25);
          } else {
            perror("Error al crear hijo2");
            exit(1);
          }
        } else {
          perror("Error al crear hijo1");
          exit(1);
        }
      }
    }

    int main(int argc, char *argv[]) {
      // 1. Validar que se haya pasado un argumento
      if (argc != 2) {
        fprintf(stderr, "Error: Debes proporcionar la altura del arbol.\n");
        fprintf(stderr, "Uso: %s <altura>\n", argv[0]);
        return 1;
      }

      // 2. Convertir el argumento a entero
      int max_nivel = atoi(argv[1]);

      // 3. Validar el rango: mayor a 4 y menor a 10 (es decir: 5, 6, 7, 8, 9)
      if (max_nivel <= 3 || max_nivel >= 10) {
        fprintf(stderr, "Error: La altura debe ser mayor a 4 y menor a 10.\n");
        return 1;
      }

      printf("Iniciando arbol de procesos con altura: %d\n", max_nivel);
      printf("--------------------------------------------------\n");

      // Llamada inicial (nivel 0)
      crear_arbol(0, max_nivel);

      return 0;
    }

Lo que aprendi y como se podria mejorar

Aprendi que fork() duplica el proceso y que waitpid() ayuda a controlar el orden de terminacion.

Se podria mejorar reemplazando sleep(25) por una espera explicita menos costosa, agregando validaciones de argc y mostrando el arbol con una identacion mas configurable.

2.3 Practica 1: Factorial padre-hijo con pipe

Codigo Fuente

Nota: Este codigo es de las notas de la materia.

#include <stdio.h>
    #include <stdlib.h>
    #include <unistd.h>
    #include <sys/wait.h>

    unsigned long long factorial(int n) {
      if(n == 0 || n == 1)
        return 1;
      return (unsigned long long)n * factorial(n - 1);
    }

    int main(int argc, char *argv[]) {
      // 1. Validamos que el usuario haya pasado exactamente 2 numeros
      // argc debe ser 3: [0] nombre del programa, [1] num1, [2] num2
      if (argc != 3) {
        fprintf(stderr, "Uso: %s <num1> <num2>\n", argv[0]);
        return 1;
      }

      // 2. Convertimos los argumentos de string a entero usando atoi()
      int num1 = atoi(argv[1]);
      int num2 = atoi(argv[2]);

      int pipefd[2];
      if (pipe(pipefd) == -1) {
        perror("Error en pipe");
        return 1;
      }

      pid_t pid = fork();

      if (pid < 0) {
        perror("Error en fork");
        return 1;
      }

      if(pid == 0){
        // HIJO
        close(pipefd[1]); // cerrar escritura

        int numero_hijo;
        read(pipefd[0], &numero_hijo, sizeof(numero_hijo));

        unsigned long long resultado_hijo = factorial(numero_hijo);
        printf("Hijo calculo: %d! = %llu\n", numero_hijo, resultado_hijo);

        close(pipefd[0]);
        exit(0);
      }
      else {
        // PADRE
        close(pipefd[0]);

        // El padre le envia el segundo numero al hijo por el pipe
        write(pipefd[1], &num2, sizeof(num2));
        close(pipefd[1]);

        unsigned long long resultado_padre = factorial(num1);

        // Esperamos a que el hijo termine para que no haya desorden en la terminal
        wait(NULL);

        printf("Padre calculo: %d! = %llu\n", num1, resultado_padre);
      }

      return 0;
    }

Salida en Terminal

user@utm:~$ gcc -Wall factorial_padre_hijo.c -o factorial_padre_hijo
    user@utm:~$ ./factorial_padre_hijo 5 6
    Hijo calculo: 6! = 720
    Padre calculo: 5! = 120

Lo que aprendi y como se podria mejorar

Aprendi que un pipe permite pasar datos del padre al hijo sin usar archivos temporales.

Se podria mejorar validando mejor los argumentos, comprobando el valor devuelto por read() y write() y usando enteros mas grandes o control de overflow para factoriales altos.

2.4 Sistema de llamado para identificar procesos

Codigo Fuente

Nota: Este codigo es de las notas de la materia.

#include <stdio.h>
#include <stdlib.h>
#include <sys/types.h>
    #include <unistd.h>

    int main(void) {
      pid_t hijo;

      hijo = fork();

      if (hijo == 0) {
        /* Codigo ejecutado por el proceso hijo */
        fprintf(stdout, "Soy el hijo, PID=%ld\n", (long)getpid());
      } else if (hijo > 0) {
        /* Codigo ejecutado por el proceso padre */
        fprintf(stdout, "Soy el padre, PID=%ld\n", (long)getpid());
      } else {
        /* Error al crear el proceso */
        perror("fork");
        return EXIT_FAILURE;
      }

      return EXIT_SUCCESS;
}

Salida en Terminal

    user@utm:~$ ./identificar_procesos
Soy el padre, PID=23697
Soy el hijo, PID=23698

Lo que aprendi y como se podria mejorar

Aprendi que fork() devuelve dos flujos de ejecucion distintos y que cada uno puede identificar su PID.

Se podria mejorar imprimiendo tambien el PID del padre e hijo con un formato uniforme y manejando mejor los errores de fork().

2.5 Sistema de llamada wait()

Codigo Fuente

Nota: Este codigo es de las notas de la materia.

#include <sys/types.h>
    #include <sys/wait.h>
    #include <unistd.h>
    #include <stdio.h>
    #include <stdlib.h>

    int main()
    {
      pid_t hijo;
      int estado;

      if ( (hijo=fork()) == -1)
      {
        perror ("fallo el fork");
        exit (EXIT_FAILURE);
      }
      else if (hijo == 0)
        fprintf (stderr, "soy el hijo con pid = %ld \n", (long) getpid());
      else if (wait(&estado) != hijo)
        fprintf (stderr, "una señal debio interrumpir la espera \n");
      else
        fprintf (stderr, "soy el padre con pid = %ld e hijo con pid = %ld\n",
             (long)getpid(), (long)hijo);

      exit(EXIT_SUCCESS);
    }

soy el padre con pid = 12909 e hijo con pid = 12910

Lo que aprendi y como se podria mejorar

Aprendi que wait() bloquea al padre hasta que el hijo termina y que sirve para evitar procesos huerfanos o salidas desordenadas.

Se podria mejorar leyendo el estado con macros como WIFEXITED y WEXITSTATUS, ademas de separar mejor el manejo de errores.

#include <stdlib.h>
#include <sys/types.h>
#include <unistd.h>

int main(void) {
    pid_t pid;
    pid = fork();

    if (pid == 0) { /* Proceso hijo */
        printf("Hijo terminado. PID=%ld\n", (long)getpid());
        exit(EXIT_SUCCESS);
    } else { /* Proceso padre */
        printf("Padre en ejecucion. PID=%ld\n", (long)getpid());

    return EXIT_SUCCESS;
}

2.8 Hilos

A diferencia de los procesos, los hilos comparten el mismo espacio de memoria del proceso padre. Son mas ligeros y utiles para tareas concurrentes dentro de una misma aplicacion.

2.8.2 Creacion de hilos

Codigo Fuente

Nota: Este codigo es de las notas de la materia.

#include <pthread.h>
    #include <stdio.h>
    #include <stdlib.h>

    typedef struct dhilos {
      int id;
      long prod;
    } DHILOS;

    DHILOS pm_hilos[10];

    void *factorial(void *valor);

    int main(int argc, char *argv[])
    {
      pthread_t tid[argc - 1];
      pthread_attr_t attr;
      int i;
      if (argc < 2)
      {
        perror("Uso: ./chilos <entero1> <entero2> ... \n");
        exit(1);
      }
      // Crear hilos
      for (i = 0; i < argc - 1; i++)
      {
        pthread_attr_init(&attr); // Atributos predeterminados
        pm_hilos[i].id = i + 1;
        pm_hilos[i].prod = atol(argv[i + 1]);
        pthread_create(&tid[i], &attr, factorial, &pm_hilos[i]);
      }
      // espera el proceso principal a cada hilo creado
      for (i = 0; i < argc - 1; i++)
        pthread_join(tid[i], NULL); // Esperar finalizacion, semejante a wait-cuando usa fork
      for (i = 0; i < argc - 1; i++)
        printf("Hilo-%d calculo Factorial de: %s = %ld\n", pm_hilos[i].id, argv[i + 1], pm_hilos[i].prod);
      return EXIT_SUCCESS;
    }

    void *factorial(void *valor)
    {
      int i = 1, prod = 1;
      DHILOS *datos;
      datos = (DHILOS *)valor;
      while (i <= datos->prod)
        prod *= (i++);
      datos->prod = prod;
      pthread_exit(&(datos->prod));
    }

Salida en Terminal

user@utm:~$ gcc -Wall chilos.c -lpthread -o chilos
    user@utm:~$ ./chilos 3 4 5
    Hilo-1 calculo Factorial de: 3 = 6
    Hilo-2 calculo Factorial de: 4 = 24
    Hilo-3 calculo Factorial de: 5 = 120

Lo que aprendi y como se podria mejorar

Aprendi que los hilos comparten memoria y que pthread_join() permite sincronizar su finalizacion.

Se podria mejorar protegiendo datos compartidos con mutexes, evitando arreglos globales fijos y validando los retornos de pthread_create() y pthread_join().
